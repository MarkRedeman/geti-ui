# Sparkline chart (composition)

`Sparkline` is a compact, axis-free trend chart for inline usage.

## Variant — `area` off

```tsx preview
import { Sparkline } from '@geti-ai/charts';

export default function App() {
  return <Sparkline data={[72, 75, 79, 81, 83, 85, 88].map((value) => ({ value }))} area={false} />;
}
```

## Variant — `area` on

```tsx preview
import { Sparkline } from '@geti-ai/charts';

export default function App() {
  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: 12 }}>
      <Sparkline data={[72, 75, 79, 81, 83, 85, 88].map((value) => ({ value }))} area />
      <Sparkline data={[80, 82, 84, 86, 87, 89, 91].map((value) => ({ value }))} />
    </div>
  );
}
```

## Variant — `animate` on

```tsx preview
import { Sparkline } from '@geti-ai/charts';

export default function App() {
  return <Sparkline data={[60, 65, 70, 74, 80, 84, 90].map((value) => ({ value }))} animate />;
}
```

## Variant — categorical mini-trends

```tsx preview
import { Sparkline } from '@geti-ai/charts';

const trends = {
  car: [51, 52, 53, 54, 55, 56, 56],
  person: [63, 62, 61, 63, 64, 65, 66],
  bike: [44, 45, 44, 47, 46, 48, 49],
};

export default function App() {
  return (
    <div style={{ display: 'grid', gap: 10 }}>
      {Object.entries(trends).map(([label, values]) => (
        <div key={label} style={{ display: 'flex', alignItems: 'center', gap: 10 }}>
          <span style={{ width: 70, textTransform: 'capitalize' }}>{label}</span>
          <Sparkline data={values.map((value) => ({ value }))} area />
        </div>
      ))}
    </div>
  );
}
```
