# Meter chart (composition)

`MeterChart` is a half-donut score indicator with a centered value label.

## Variant — default

```tsx preview
import { MeterChart } from '@geti-ai/charts';

export default function App() {
  return <MeterChart value={84} aria-label="Model quality score" height={220} />;
}
```

## Variant — custom max and formatter

```tsx preview
import { MeterChart } from '@geti-ai/charts';

export default function App() {
  return (
    <MeterChart
      value={0.873}
      max={1}
      valueFormatter={(v) => `mAP ${v.toFixed(3)}`}
      aria-label="mAP meter"
      height={220}
    />
  );
}
```

## Variant — custom colors

```tsx preview
import { MeterChart } from '@geti-ai/charts';

export default function App() {
  return (
    <MeterChart
      value={63}
      color="var(--spectrum-global-color-orange-500)"
      trackColor="var(--spectrum-global-color-gray-700)"
      aria-label="Risk score meter"
      height={220}
    />
  );
}
```

## Variant — train / validation / test palette

```tsx preview
import { MeterChart, getDatasetSubsetColor } from '@geti-ai/charts';

export default function App() {
  return (
    <div style={{ display: 'grid', gridTemplateColumns: 'repeat(3, minmax(0, 1fr))', gap: 12 }}>
      <MeterChart value={91} color={getDatasetSubsetColor('train')} aria-label="Train meter" height={210} />
      <MeterChart value={86} color={getDatasetSubsetColor('validation')} aria-label="Validation meter" height={210} />
      <MeterChart value={84} color={getDatasetSubsetColor('test')} aria-label="Test meter" height={210} />
    </div>
  );
}
```
