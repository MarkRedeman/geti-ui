# Precision recall curve (primitive)

Use Recharts primitives when you need full control over PR curves, baselines, and annotations.

## Variant — single PR curve

```tsx preview
import { Line, LineChart } from 'recharts';
import {
  GetiCartesianGrid,
  GetiLegend,
  GetiResponsiveContainer,
  GetiTooltip,
  GetiXAxis,
  GetiYAxis,
} from '@geti-ai/charts/primitives';

const data = [
  { recall: 0.0, precision: 1.0 },
  { recall: 0.1, precision: 0.98 },
  { recall: 0.2, precision: 0.95 },
  { recall: 0.35, precision: 0.91 },
  { recall: 0.5, precision: 0.87 },
  { recall: 0.65, precision: 0.80 },
  { recall: 0.8, precision: 0.69 },
  { recall: 0.9, precision: 0.55 },
  { recall: 1.0, precision: 0.42 },
];

export default function App() {
  return (
    <GetiResponsiveContainer height={320}>
      <LineChart data={data}>
        <GetiCartesianGrid />
        <GetiXAxis type="number" dataKey="recall" domain={[0, 1]} />
        <GetiYAxis type="number" domain={[0, 1]} />
        <GetiTooltip />
        <GetiLegend />

        <Line dataKey="precision" name="Detector A" stroke="var(--spectrum-global-color-blue-500)" dot={false} isAnimationActive={false} />
      </LineChart>
    </GetiResponsiveContainer>
  );
}
```

## Variant — two models + no-skill baseline

```tsx preview
import { Line, LineChart, ReferenceLine } from 'recharts';
import {
  GetiCartesianGrid,
  GetiLegend,
  GetiResponsiveContainer,
  GetiTooltip,
  GetiXAxis,
  GetiYAxis,
} from '@geti-ai/charts/primitives';

const data = [
  { recall: 0.0, modelA: 1.0, modelB: 1.0 },
  { recall: 0.1, modelA: 0.98, modelB: 0.96 },
  { recall: 0.2, modelA: 0.95, modelB: 0.92 },
  { recall: 0.35, modelA: 0.91, modelB: 0.86 },
  { recall: 0.5, modelA: 0.87, modelB: 0.79 },
  { recall: 0.65, modelA: 0.80, modelB: 0.71 },
  { recall: 0.8, modelA: 0.69, modelB: 0.60 },
  { recall: 0.9, modelA: 0.55, modelB: 0.48 },
  { recall: 1.0, modelA: 0.42, modelB: 0.38 },
];

export default function App() {
  return (
    <GetiResponsiveContainer height={320}>
      <LineChart data={data}>
        <GetiCartesianGrid />
        <GetiXAxis type="number" dataKey="recall" domain={[0, 1]} />
        <GetiYAxis type="number" domain={[0, 1]} />
        <GetiTooltip />
        <GetiLegend />

        <ReferenceLine y={0.38} stroke="var(--spectrum-global-color-gray-500)" strokeDasharray="4 3" />

        <Line dataKey="modelA" name="Detector A" stroke="var(--spectrum-global-color-blue-500)" dot={false} isAnimationActive={false} />
        <Line dataKey="modelB" name="Detector B" stroke="var(--spectrum-global-color-purple-500)" dot={false} isAnimationActive={false} />
      </LineChart>
    </GetiResponsiveContainer>
  );
}
```
