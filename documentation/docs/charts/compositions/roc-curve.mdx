# ROC curve (composition)

`RocCurve` is an ML composition for visualizing true-positive-rate vs false-positive-rate trade-offs.

## Props

| Prop | Type | Default | Description |
|---|---|---|---|
| `data` | `object[]` | required | ROC rows with FPR + one/more TPR keys |
| `series` | `RocCurveSeriesConfig[]` | required | ROC curves to draw |
| `xAxisKey` | `string` | `'fpr'` | False-positive-rate key |
| `showBaseline` | `boolean` | `true` | Show random-classifier diagonal |
| `showLegend` | `boolean` | `true` | Show legend |
| `showTooltip` | `boolean` | `true` | Show tooltip |
| `animate` | `boolean` | `false` | Opt-in animation |

## Variant â€” single model ROC

```tsx preview
import { RocCurve } from '@geti-ui/charts';

const data = [
  { fpr: 0.0, tpr: 0.0 },
  { fpr: 0.02, tpr: 0.28 },
  { fpr: 0.05, tpr: 0.48 },
  { fpr: 0.10, tpr: 0.66 },
  { fpr: 0.20, tpr: 0.81 },
  { fpr: 0.35, tpr: 0.90 },
  { fpr: 0.55, tpr: 0.96 },
  { fpr: 1.0, tpr: 1.0 },
];

export default function App() {
  return (
    <RocCurve
      data={data}
      series={[{ dataKey: 'tpr', name: 'Detector A' }]}
      aria-label="ROC curve for detector A"
      height={320}
    />
  );
}
```

## Variant â€” compare two models

```tsx preview
import { RocCurve } from '@geti-ui/charts';

const data = [
  { fpr: 0.0, modelA: 0.0, modelB: 0.0 },
  { fpr: 0.02, modelA: 0.28, modelB: 0.21 },
  { fpr: 0.05, modelA: 0.48, modelB: 0.39 },
  { fpr: 0.10, modelA: 0.66, modelB: 0.54 },
  { fpr: 0.20, modelA: 0.81, modelB: 0.70 },
  { fpr: 0.35, modelA: 0.90, modelB: 0.84 },
  { fpr: 0.55, modelA: 0.96, modelB: 0.93 },
  { fpr: 1.0, modelA: 1.0, modelB: 1.0 },
];

export default function App() {
  return (
    <RocCurve
      data={data}
      series={[
        { dataKey: 'modelA', name: 'Detector A' },
        { dataKey: 'modelB', name: 'Detector B' },
      ]}
      aria-label="ROC comparison for two models"
      height={320}
    />
  );
}
```

## Variant â€” hide baseline + percent axes

```tsx preview
import { RocCurve } from '@geti-ui/charts';

const data = [
  { fpr: 0.0, tpr: 0.0 },
  { fpr: 0.02, tpr: 0.28 },
  { fpr: 0.05, tpr: 0.48 },
  { fpr: 0.10, tpr: 0.66 },
  { fpr: 0.20, tpr: 0.81 },
  { fpr: 0.35, tpr: 0.90 },
  { fpr: 0.55, tpr: 0.96 },
  { fpr: 1.0, tpr: 1.0 },
];

const percent = (value: number) => `${Math.round(value * 100)}%`;

export default function App() {
  return (
    <RocCurve
      data={data}
      series={[{ dataKey: 'tpr', name: 'Detector A' }]}
      showBaseline={false}
      xAxisProps={{ tickFormatter: percent }}
      yAxisProps={{ tickFormatter: percent }}
      aria-label="ROC curve with percent axis labels"
      height={320}
    />
  );
}
```
