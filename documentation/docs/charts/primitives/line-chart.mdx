# Line chart (primitive)

Use Recharts `LineChart` with Geti primitives when you need custom low-level control.

## Variant — no legend

```tsx preview
import { Line, LineChart } from 'recharts';
import {
  GetiCartesianGrid,
  GetiResponsiveContainer,
  GetiTooltip,
  GetiXAxis,
  GetiYAxis,
} from '@geti-ai/charts/primitives';

const data = [
  { epoch: 1, loss: 1.42 },
  { epoch: 2, loss: 1.10 },
  { epoch: 3, loss: 0.87 },
  { epoch: 4, loss: 0.71 },
];

export default function App() {
  return (
    <GetiResponsiveContainer height={260}>
      <LineChart data={data}>
        <GetiCartesianGrid />
        <GetiXAxis dataKey="epoch" />
        <GetiYAxis />
        <GetiTooltip />
        <Line type="monotone" dataKey="loss" stroke="var(--spectrum-global-color-blue-500)" dot={false} isAnimationActive={false} />
      </LineChart>
    </GetiResponsiveContainer>
  );
}
```

## Variant — time-based axis + custom tick formatter

```tsx preview
import { Line, LineChart } from 'recharts';
import {
  GetiCartesianGrid,
  GetiResponsiveContainer,
  GetiTooltip,
  GetiXAxis,
  GetiYAxis,
} from '@geti-ai/charts/primitives';

const data = [
  { ts: Date.parse('2026-01-01T00:00:00Z'), qps: 220 },
  { ts: Date.parse('2026-01-02T00:00:00Z'), qps: 280 },
  { ts: Date.parse('2026-01-03T00:00:00Z'), qps: 310 },
  { ts: Date.parse('2026-01-04T00:00:00Z'), qps: 260 },
  { ts: Date.parse('2026-01-05T00:00:00Z'), qps: 340 },
];

export default function App() {
  return (
    <GetiResponsiveContainer height={260}>
      <LineChart data={data}>
        <GetiCartesianGrid />
        <GetiXAxis
          dataKey="ts"
          type="number"
          scale="time"
          domain={['dataMin', 'dataMax']}
          tickFormatter={(v) => new Date(Number(v)).toLocaleDateString('en-US', { month: 'short', day: 'numeric' })}
        />
        <GetiYAxis />
        <GetiTooltip />
        <Line type="monotone" dataKey="qps" stroke="var(--spectrum-global-color-blue-500)" dot={false} isAnimationActive={false} />
      </LineChart>
    </GetiResponsiveContainer>
  );
}
```

## Variant — interpolation methods

```tsx preview
import { Line, LineChart } from 'recharts';
import {
  GetiCartesianGrid,
  GetiLegend,
  GetiResponsiveContainer,
  GetiTooltip,
  GetiXAxis,
  GetiYAxis,
} from '@geti-ai/charts/primitives';

const data = [
  { epoch: 1, linear: 1.42, natural: 1.42, step: 1.42 },
  { epoch: 2, linear: 1.10, natural: 1.10, step: 1.10 },
  { epoch: 3, linear: 0.87, natural: 0.87, step: 0.87 },
  { epoch: 4, linear: 0.71, natural: 0.71, step: 0.71 },
  { epoch: 5, linear: 0.60, natural: 0.60, step: 0.60 },
];

export default function App() {
  return (
    <GetiResponsiveContainer height={280}>
      <LineChart data={data}>
        <GetiCartesianGrid />
        <GetiXAxis dataKey="epoch" />
        <GetiYAxis />
        <GetiTooltip />
        <GetiLegend />
        <Line type="linear" dataKey="linear" stroke="var(--spectrum-global-color-blue-500)" dot={false} isAnimationActive={false} />
        <Line type="natural" dataKey="natural" stroke="var(--spectrum-global-color-green-500)" dot={false} isAnimationActive={false} />
        <Line type="stepAfter" dataKey="step" stroke="var(--spectrum-global-color-yellow-500)" dot={false} isAnimationActive={false} />
      </LineChart>
    </GetiResponsiveContainer>
  );
}
```

## Variant — dataset subset line colors

```tsx preview
import { Line, LineChart } from 'recharts';
import {
  GetiCartesianGrid,
  GetiLegend,
  GetiResponsiveContainer,
  GetiTooltip,
  GetiXAxis,
  GetiYAxis,
} from '@geti-ai/charts/primitives';
import { getDatasetSubsetColor } from '@geti-ai/charts';

const data = [
  { epoch: 1, train: 1.42, val: 1.50, test: 1.55 },
  { epoch: 2, train: 1.10, val: 1.20, test: 1.28 },
  { epoch: 3, train: 0.87, val: 0.98, test: 1.06 },
  { epoch: 4, train: 0.71, val: 0.82, test: 0.90 },
];

export default function App() {
  return (
    <GetiResponsiveContainer height={280}>
      <LineChart data={data}>
        <GetiCartesianGrid />
        <GetiXAxis dataKey="epoch" />
        <GetiYAxis />
        <GetiTooltip />
        <GetiLegend />
        <Line type="monotone" dataKey="train" stroke={getDatasetSubsetColor('train')} dot={false} isAnimationActive={false} />
        <Line type="monotone" dataKey="val" stroke={getDatasetSubsetColor('val')} dot={false} isAnimationActive={false} />
        <Line type="monotone" dataKey="test" stroke={getDatasetSubsetColor('test')} dot={false} isAnimationActive={false} />
      </LineChart>
    </GetiResponsiveContainer>
  );
}
```
