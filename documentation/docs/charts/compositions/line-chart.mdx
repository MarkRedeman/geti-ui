# Line chart (composition)

`LineChart` is a ready-to-use composition. It wires responsive container, axes, grid, tooltip, and series rendering for you.

## Default (boolean props off)

```tsx preview
import { LineChart } from '@geti-ai/charts';

const data = [
  { epoch: 1, train: 1.42, val: 1.50 },
  { epoch: 2, train: 1.10, val: 1.20 },
  { epoch: 3, train: 0.87, val: 0.98 },
  { epoch: 4, train: 0.71, val: 0.82 },
];

export default function App() {
  return (
    <LineChart
      data={data}
      xAxisKey="epoch"
      series={[
        { dataKey: 'train', name: 'Train loss' },
        { dataKey: 'val', name: 'Val loss' },
      ]}
      aria-label="Training and validation loss"
      height={260}
    />
  );
}
```

## Variant — train / validation / test palette

```tsx preview
import { LineChart, withDatasetSubsetPalette } from '@geti-ai/charts';

const data = [
  { epoch: 1, train: 1.42, val: 1.50, test: 1.55 },
  { epoch: 2, train: 1.10, val: 1.20, test: 1.28 },
  { epoch: 3, train: 0.87, val: 0.98, test: 1.06 },
  { epoch: 4, train: 0.71, val: 0.82, test: 0.90 },
];

export default function App() {
  const series = withDatasetSubsetPalette([
    { dataKey: 'train', name: 'Training' },
    { dataKey: 'val', name: 'Validation' },
    { dataKey: 'test', name: 'Test' },
  ]);

  return (
    <LineChart
      data={data}
      xAxisKey="epoch"
      showLegend
      series={series}
      aria-label="Training/validation/test loss curve"
      height={260}
    />
  );
}
```

## Variant — time-based X axis

```tsx preview
import { LineChart } from '@geti-ai/charts';

const data = [
  { ts: Date.parse('2026-01-01T00:00:00Z'), throughput: 22 },
  { ts: Date.parse('2026-01-02T00:00:00Z'), throughput: 28 },
  { ts: Date.parse('2026-01-03T00:00:00Z'), throughput: 31 },
  { ts: Date.parse('2026-01-04T00:00:00Z'), throughput: 26 },
  { ts: Date.parse('2026-01-05T00:00:00Z'), throughput: 34 },
];

export default function App() {
  return (
    <LineChart
      data={data}
      xAxisKey="ts"
      xScale={{ scale: 'time' }}
      xAxisProps={{
        type: 'number',
        domain: ['dataMin', 'dataMax'],
        tickFormatter: (v) => new Date(Number(v)).toLocaleDateString('en-US', { month: 'short', day: 'numeric' }),
      }}
      series={[{ dataKey: 'throughput', name: 'Images/sec' }]}
      aria-label="Throughput over time"
      height={260}
    />
  );
}
```
