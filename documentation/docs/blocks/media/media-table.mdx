# Media table

`MediaTable` presents media as one item per table row, with a thumbnail column enabled by default.

Use it when users need media metadata scanning, sorting, and bulk actions in a denser layout than a thumbnail grid.

If you need custom row composition, provide `EntryComponent`. The default implementation is exported as `MediaEntry`.

```tsx
import {
  MediaTable,
  MediaEntry,
  type MediaTableColumn,
  type MediaEntryProps,
  type MediaTableRenderContext,
  type MediaGridSelection,
} from '@geti-ai/blocks';
```

## Basic usage

```tsx preview
import { useMemo, useState } from 'react';
import { ThemeProvider, Text } from '@geti-ai/ui';
import {
  MediaTable,
  type MediaGridSelection,
  type MediaTableColumn,
  type MediaTableRenderContext,
} from '@geti-ai/blocks';

type Media = {
  id: string;
  name: string;
  kind: 'image' | 'video';
  width: number;
  height: number;
  uploadDate: string;
  thumbnailUrl: string;
};

const allMedia: Media[] = Array.from({ length: 120 }, (_, index) => ({
  id: `media-${index + 1}`,
  name: `Asset ${index + 1}`,
  kind: index % 5 === 0 ? 'video' : 'image',
  width: 1280 + (index % 4) * 160,
  height: 720 + (index % 3) * 120,
  uploadDate: `2026-02-${String((index % 27) + 1).padStart(2, '0')}`,
  thumbnailUrl: `https://picsum.photos/seed/media-table-${index + 1}/300/300`,
}));

export default function App() {
  const [loadedCount, setLoadedCount] = useState(40);
  const [selectedKeys, setSelectedKeys] = useState<MediaGridSelection>(new Set());

  const byIndex = useMemo(() => {
    const map = new Map<number, Media>();
    for (let i = 0; i < loadedCount; i += 1) {
      map.set(i, allMedia[i]);
    }
    return map;
  }, [loadedCount]);

  const columns: MediaTableColumn<Media>[] = [
    {
      key: 'name',
      name: 'Name',
      isRowHeader: true,
      allowsSorting: true,
      renderCell: (ctx: MediaTableRenderContext<Media>) => (
        <Text>{ctx.item?.name ?? 'Loading…'}</Text>
      ),
      textValue: (ctx: MediaTableRenderContext<Media>) => ctx.item?.name ?? 'Loading',
    },
    {
      key: 'kind',
      name: 'Type',
      allowsSorting: true,
      renderCell: (ctx: MediaTableRenderContext<Media>) => (
        <Text>{ctx.item?.kind ?? '—'}</Text>
      ),
    },
    {
      key: 'resolution',
      name: 'Resolution',
      renderCell: (ctx: MediaTableRenderContext<Media>) => (
        <Text>{ctx.item ? `${ctx.item.width} × ${ctx.item.height}` : '—'}</Text>
      ),
    },
    {
      key: 'date',
      name: 'Uploaded',
      allowsSorting: true,
      renderCell: (ctx: MediaTableRenderContext<Media>) => (
        <Text>{ctx.item?.uploadDate ?? '—'}</Text>
      ),
    },
  ];

  return (
    <ThemeProvider>
      <div style={{ height: 520 }}>
        <MediaTable
          totalItems={allMedia.length}
          getItemAt={(index) => byIndex.get(index)}
          columns={columns}
          selectionMode="multiple"
          selectedKeys={selectedKeys}
          onSelectionChange={setSelectedKeys}
          thumbnailSize={50}
          onSortChange={(descriptor) => {
            console.log('sort change', descriptor.column, descriptor.direction);
          }}
          onItemPress={(context) => {
            if (context.item && context.index >= loadedCount - 1 && loadedCount < allMedia.length) {
              setLoadedCount((prev) => Math.min(prev + 20, allMedia.length));
            }
          }}
        />
      </div>
    </ThemeProvider>
  );
}
```

## Filter + selection + `ActionBar` overview

This example mirrors the media overview filter flow and shows `ActionBar` activation when table rows are selected.

```tsx preview
import { useMemo, useState, useCallback } from 'react';
import {
  ActionBar,
  ActionBarContainer,
  ActionBarItem,
  DialogTrigger,
  Flex,
  SearchField,
  Text,
  ThemeProvider,
} from '@geti-ai/ui';
import {
  FilterChips,
  FilterDialog,
  FilterTriggerButton,
  MediaTable,
  type FilterFieldOption,
  type FilterModel,
  type FilterOperatorOption,
  type FilterRule,
  type MediaGridSelection,
  type MediaTableColumn,
  type MediaTableRenderContext,
} from '@geti-ai/blocks';

type Media = {
  id: string;
  name: string;
  kind: 'image' | 'video';
  width: number;
  height: number;
  uploadDate: string;
  labelCount: number;
  thumbnailUrl: string;
};

const media: Media[] = Array.from({ length: 160 }, (_, index) => ({
  id: `media-${index + 1}`,
  name: `Road Scene ${index + 1}`,
  kind: index % 7 === 0 ? 'video' : 'image',
  width: 1280 + (index % 4) * 160,
  height: 720 + (index % 3) * 120,
  uploadDate: `2026-03-${String((index % 27) + 1).padStart(2, '0')}`,
  labelCount: (index % 8) + 1,
  thumbnailUrl: `https://picsum.photos/seed/media-table-overview-${index + 1}/300/300`,
}));

const fields: FilterFieldOption[] = [
  { key: 'MEDIA_NAME', label: 'Media name', valueType: 'text' },
  { key: 'MEDIA_WIDTH', label: 'Width', valueType: 'number' },
  { key: 'UPLOAD_DATE', label: 'Upload date', valueType: 'date' },
  {
    key: 'MEDIA_TYPE',
    label: 'Media type',
    valueType: 'enum',
    operators: [
      { key: 'EQUAL', label: 'is' },
      { key: 'NOT_EQUAL', label: 'is not' },
    ],
    editorConfig: {
      options: [
        { key: 'image', label: 'Image' },
        { key: 'video', label: 'Video' },
      ],
    },
  },
];

const globalOperators: FilterOperatorOption[] = [
  { key: 'CONTAINS', label: 'contains' },
  { key: 'EQUAL', label: 'equal' },
  { key: 'NOT_EQUAL', label: 'not equal' },
  { key: 'GREATER', label: 'greater than' },
  { key: 'LESS', label: 'less than' },
];

const seedRules: FilterRule[] = [
  { id: '1', field: 'MEDIA_NAME', operator: 'CONTAINS', value: 'Road' },
  { id: '2', field: 'MEDIA_WIDTH', operator: 'GREATER', value: 1280 },
];

function selectionCount(selection: MediaGridSelection) {
  return selection === 'all' ? 'all' : selection.size;
}

export default function App() {
  const [applied, setApplied] = useState<FilterModel>({ condition: 'and', rules: seedRules });
  const [selectedKeys, setSelectedKeys] = useState<MediaGridSelection>(new Set());
  const [searchText, setSearchText] = useState('');

  const handleRemoveRule = useCallback((ruleId: string) => {
    setApplied((prev) => ({
      ...prev,
      rules: prev.rules.filter((rule) => rule.id !== ruleId),
    }));
  }, []);

  const handleClearAll = useCallback(() => {
    setApplied((prev) => ({ ...prev, rules: [] }));
  }, []);

  const filteredMedia = useMemo(() => {
    if (!searchText.trim()) {
      return media;
    }

    const query = searchText.toLowerCase();
    return media.filter((item) => item.name.toLowerCase().includes(query));
  }, [searchText]);

  const byIndex = useMemo(() => {
    const map = new Map<number, Media>();
    filteredMedia.forEach((item, index) => map.set(index, item));
    return map;
  }, [filteredMedia]);

  const columns: MediaTableColumn<Media>[] = [
    {
      key: 'name',
      name: 'Name',
      isRowHeader: true,
      allowsSorting: true,
      renderCell: (ctx: MediaTableRenderContext<Media>) => (
        <Text>{ctx.item?.name ?? 'Loading…'}</Text>
      ),
      textValue: (ctx: MediaTableRenderContext<Media>) => ctx.item?.name ?? 'Loading',
    },
    {
      key: 'kind',
      name: 'Type',
      allowsSorting: true,
      renderCell: (ctx: MediaTableRenderContext<Media>) => (
        <Text>{ctx.item?.kind ?? '—'}</Text>
      ),
    },
    {
      key: 'labels',
      name: 'Labels',
      allowsSorting: true,
      renderCell: (ctx: MediaTableRenderContext<Media>) => (
        <Text>{ctx.item ? String(ctx.item.labelCount) : '—'}</Text>
      ),
    },
    {
      key: 'date',
      name: 'Uploaded',
      allowsSorting: true,
      renderCell: (ctx: MediaTableRenderContext<Media>) => (
        <Text>{ctx.item?.uploadDate ?? '—'}</Text>
      ),
    },
  ];

  const count = selectionCount(selectedKeys);

  return (
    <ThemeProvider>
      <Flex direction="column" gap="size-200">
        <Flex alignItems="center" gap="size-100">
          <DialogTrigger type="popover" hideArrow>
            <FilterTriggerButton
              isSelected={applied.rules.length > 0}
              ariaLabel="Filter media"
            />
            <FilterDialog
              fields={fields}
              globalOperators={globalOperators}
              defaultDraft={applied}
              onApply={setApplied}
              totalMatches={filteredMedia.length}
            />
          </DialogTrigger>

          <SearchField
            aria-label="Search by file name"
            placeholder="Search by file name"
            value={searchText}
            onChange={setSearchText}
            onClear={() => setSearchText('')}
            width="size-3400"
          />
        </Flex>

        {applied.rules.length > 0 ? (
          <FilterChips
            rules={applied.rules}
            fields={fields}
            onRemoveRule={handleRemoveRule}
            onClearAll={handleClearAll}
          />
        ) : null}

        <div style={{ height: 520 }}>
          <ActionBarContainer height="100%">
            <MediaTable
              totalItems={filteredMedia.length}
              getItemAt={(index) => byIndex.get(index)}
              columns={columns}
              selectionMode="multiple"
              selectedKeys={selectedKeys}
              onSelectionChange={setSelectedKeys}
              thumbnailSize={50}
            />

            <ActionBar
              selectedItemCount={count}
              onClearSelection={() => setSelectedKeys(new Set())}
              onAction={(key) => console.log('action', key)}
            >
              <ActionBarItem key="annotate">Annotate</ActionBarItem>
              <ActionBarItem key="download">Download</ActionBarItem>
              <ActionBarItem key="delete">Delete</ActionBarItem>
            </ActionBar>
          </ActionBarContainer>
        </div>
      </Flex>
    </ThemeProvider>
  );
}
```

## `MediaTable` props

| Prop | Type | Required | Description |
| --- | --- | --- | --- |
| `totalItems` | `number` | Yes | Total number of rows represented by the table. |
| `getItemAt` | `(index: number) => T \| undefined` | Yes | Returns row item data by absolute index, or `undefined` for placeholder rows. |
| `columns` | `MediaTableColumn<T>[]` | Yes | Declarative column definitions for non-thumbnail columns. |
| `selectionMode` | `'none' \| 'single' \| 'multiple'` | No | Selection behavior. Default: `'multiple'`. |
| `selectedKeys` | `Set<string> \| 'all'` | No | Controlled row selection state. |
| `defaultSelectedKeys` | `Iterable<string>` | No | Initial uncontrolled row selection state. |
| `onSelectionChange` | `(keys: Set<string> \| 'all') => void` | No | Called when selection changes. |
| `focusOnSelect` | `boolean` | No | Auto-focus/scroll selected row into view when selection changes. Default: `false`. |
| `isLoading` | `boolean` | No | Keeps table scaffold visible while data is loading. |
| `emptyState` | `ReactNode` | No | Custom empty-state UI. |
| `ariaLabel` | `string` | No | Accessible table label. Default: `'Media table'`. |
| `thumbnailSize` | `number` | No | Thumbnail width/height in px. Default: `50`. |
| `thumbnailColumnKey` | `string` | No | Internal key for the auto thumbnail column. |
| `thumbnailColumnHeader` | `string` | No | Header text for the thumbnail column. Default: `'Thumbnail'`. |
| `hideThumbnailColumn` | `boolean` | No | When `true`, suppresses the auto thumbnail column. |
| `EntryComponent` | `(props: MediaEntryProps<T>) => ReactElement` | No | Cell renderer override used inside each generated row. Defaults to exported `MediaEntry`. |
| `sortDescriptor` | `{ column: string \| number; direction: 'ascending' \| 'descending' }` | No | Controlled sort descriptor passed to the table. |
| `onSortChange` | `(descriptor) => void` | No | Called when a sortable column header changes sort state. |
| `density` | `'compact' \| 'regular' \| 'spacious'` | No | Table density. Default: `'regular'`. |
| `overflowMode` | `'wrap' \| 'truncate'` | No | Text overflow strategy. Defaults to `'wrap'`, allowing row height to grow with content. |
| `onItemPress` | `(context, event?) => void` | No | Called when an item press handler is triggered from row content. |
| `onItemDoublePress` | `(context) => void` | No | Called when an item double-press handler is triggered from row content. |
| `className` | `string` | No | Optional root class. |
| `style` | `CSSProperties` | No | Inline root styles. |

## `MediaTableColumn` shape

| Field | Type | Description |
| --- | --- | --- |
| `key` | `string` | Stable column key. |
| `name` | `string` | Header label text. |
| `renderCell` | `(context: MediaTableRenderContext<T>) => ReactNode` | Renders each row cell for this column. |
| `textValue` | `(context) => string` | Optional text value for accessibility/search. |
| `allowsSorting` | `boolean` | Enables sortable header behavior. |
| `isRowHeader` | `boolean` | Marks the column as row header for accessibility. |

## Integration notes

- Keep domain filtering in your own state/query layer; `MediaTable` only renders given rows.
- Combine `selectedKeys` + `ActionBar` for bulk actions.
- If your item does not expose `thumbnailUrl`/`thumbnailSrc`, use a custom `EntryComponent` to render rows.
- Reuse the same filter state model (`FilterDialog` + `FilterChips`) from the grid flow when switching views.
