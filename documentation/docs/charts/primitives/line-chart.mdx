# Line chart (primitive)

Use Recharts `LineChart` with Geti primitives when you need custom low-level control.

## Variant — no legend

```tsx preview
import { Line, LineChart } from 'recharts';
import {
  GetiCartesianGrid,
  GetiResponsiveContainer,
  GetiTooltip,
  GetiXAxis,
  GetiYAxis,
} from '@geti-ai/charts/primitives';

const data = [
  { epoch: 1, loss: 1.42 },
  { epoch: 2, loss: 1.10 },
  { epoch: 3, loss: 0.87 },
  { epoch: 4, loss: 0.71 },
];

export default function App() {
  return (
    <GetiResponsiveContainer height={260}>
      <LineChart data={data}>
        <GetiCartesianGrid />
        <GetiXAxis dataKey="epoch" />
        <GetiYAxis />
        <GetiTooltip />
        <Line type="monotone" dataKey="loss" stroke="var(--spectrum-global-color-blue-500)" dot={false} isAnimationActive={false} />
      </LineChart>
    </GetiResponsiveContainer>
  );
}
```

## Variant — legend on

```tsx preview
import { Line, LineChart } from 'recharts';
import {
  GetiCartesianGrid,
  GetiLegend,
  GetiResponsiveContainer,
  GetiTooltip,
  GetiXAxis,
  GetiYAxis,
} from '@geti-ai/charts/primitives';

const data = [
  { epoch: 1, train: 1.42, val: 1.50 },
  { epoch: 2, train: 1.10, val: 1.20 },
  { epoch: 3, train: 0.87, val: 0.98 },
  { epoch: 4, train: 0.71, val: 0.82 },
];

export default function App() {
  return (
    <GetiResponsiveContainer height={260}>
      <LineChart data={data}>
        <GetiCartesianGrid />
        <GetiXAxis dataKey="epoch" />
        <GetiYAxis />
        <GetiTooltip />
        <GetiLegend />
        <Line type="monotone" dataKey="train" stroke="var(--spectrum-global-color-blue-500)" dot={false} isAnimationActive={false} />
        <Line type="monotone" dataKey="val" stroke="var(--spectrum-global-color-green-500)" dot={false} isAnimationActive={false} />
      </LineChart>
    </GetiResponsiveContainer>
  );
}
```

## Variant — time-based axis + custom tick formatter

```tsx preview
import { Line, LineChart } from 'recharts';
import {
  GetiCartesianGrid,
  GetiResponsiveContainer,
  GetiTooltip,
  GetiXAxis,
  GetiYAxis,
} from '@geti-ai/charts/primitives';

const data = [
  { ts: Date.parse('2026-01-01T00:00:00Z'), qps: 220 },
  { ts: Date.parse('2026-01-02T00:00:00Z'), qps: 280 },
  { ts: Date.parse('2026-01-03T00:00:00Z'), qps: 310 },
  { ts: Date.parse('2026-01-04T00:00:00Z'), qps: 260 },
  { ts: Date.parse('2026-01-05T00:00:00Z'), qps: 340 },
];

export default function App() {
  return (
    <GetiResponsiveContainer height={260}>
      <LineChart data={data}>
        <GetiCartesianGrid />
        <GetiXAxis
          dataKey="ts"
          type="number"
          scale="time"
          domain={['dataMin', 'dataMax']}
          tickFormatter={(v) => new Date(Number(v)).toLocaleDateString('en-US', { month: 'short', day: 'numeric' })}
        />
        <GetiYAxis />
        <GetiTooltip />
        <Line type="monotone" dataKey="qps" stroke="var(--spectrum-global-color-blue-500)" dot={false} isAnimationActive={false} />
      </LineChart>
    </GetiResponsiveContainer>
  );
}
```

## Variant — log scale on Y axis

```tsx preview
import { Line, LineChart } from 'recharts';
import {
  GetiCartesianGrid,
  GetiResponsiveContainer,
  GetiTooltip,
  GetiXAxis,
  GetiYAxis,
} from '@geti-ai/charts/primitives';

const data = [
  { epoch: 1, loss: 1200 },
  { epoch: 2, loss: 380 },
  { epoch: 3, loss: 110 },
  { epoch: 4, loss: 42 },
  { epoch: 5, loss: 12 },
];

export default function App() {
  return (
    <GetiResponsiveContainer height={260}>
      <LineChart data={data}>
        <GetiCartesianGrid />
        <GetiXAxis dataKey="epoch" />
        <GetiYAxis scale="log" domain={[1, 'auto']} />
        <GetiTooltip />
        <Line type="monotone" dataKey="loss" stroke="var(--spectrum-global-color-blue-500)" dot={false} isAnimationActive={false} />
      </LineChart>
    </GetiResponsiveContainer>
  );
}
```
