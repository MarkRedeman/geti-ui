export function ToggleButtonsDemo() {
  const [selected, setSelected] = useState('grid');
  return (
    <ToggleButtons
      options={['grid', 'list']}
      selectedOption={selected}
      onOptionChange={setSelected}
      getLabel={(option) => {
        if (option === 'grid') return 'Grid';
        return 'List';
      }}
    />
  );
}
