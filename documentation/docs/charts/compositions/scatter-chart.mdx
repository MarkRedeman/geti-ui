# Scatter chart (composition)

`ScatterChart` is a ready-to-use composition for 2D point data, including optional Voronoi hover.

## Variant — `useVoronoi` off

```tsx preview
import { ScatterChart } from '@geti-ai/charts';

const points = [
  { x: 0.1, y: 0.85, name: 'cat' },
  { x: 0.3, y: 0.72, name: 'dog' },
  { x: 0.5, y: 0.91, name: 'car' },
  { x: 0.7, y: 0.77, name: 'bicycle' },
];

export default function App() {
  return (
    <ScatterChart
      series={[{ name: 'Per class', data: points }]}
      xAxisLabel="False positive rate"
      yAxisLabel="True positive rate"
      aria-label="Scatter without Voronoi"
      height={280}
      useVoronoi={false}
    />
  );
}
```

## Variant — `useVoronoi` on

```tsx preview
import { ScatterChart } from '@geti-ai/charts';

const points = [
  { x: 0.1, y: 0.85, name: 'cat' },
  { x: 0.3, y: 0.72, name: 'dog' },
  { x: 0.5, y: 0.91, name: 'car' },
  { x: 0.7, y: 0.77, name: 'bicycle' },
];

export default function App() {
  return (
    <ScatterChart
      series={[{ name: 'Per class', data: points }]}
      xAxisLabel="False positive rate"
      yAxisLabel="True positive rate"
      aria-label="ROC-style class scatter"
      height={280}
      useVoronoi
    />
  );
}
```

## Variant — chart booleans on (`showLegend`, `animate`)

```tsx preview
import { ScatterChart } from '@geti-ai/charts';

const trainPoints = [
  { x: 0.1, y: 0.85, name: 'cat' },
  { x: 0.3, y: 0.72, name: 'dog' },
  { x: 0.5, y: 0.91, name: 'car' },
];

const valPoints = [
  { x: 0.2, y: 0.80, name: 'cat' },
  { x: 0.4, y: 0.70, name: 'dog' },
  { x: 0.6, y: 0.88, name: 'car' },
];

export default function App() {
  return (
    <ScatterChart
      series={[
        { name: 'Train', data: trainPoints },
        { name: 'Val', data: valPoints },
      ]}
      showLegend
      animate
      useVoronoi
      xAxisLabel="False positive rate"
      yAxisLabel="True positive rate"
      aria-label="Scatter with legend and animation"
      height={280}
    />
  );
}
```

## Variant — log/log scale

```tsx preview
import { ScatterChart } from '@geti-ai/charts';

const models = [
  { x: 2e6, y: 1.2e9, name: 'M1' },
  { x: 8e6, y: 4.6e9, name: 'M2' },
  { x: 2.4e7, y: 1.1e10, name: 'M3' },
  { x: 6.4e7, y: 3.0e10, name: 'M4' },
];

export default function App() {
  return (
    <ScatterChart
      series={[{ name: 'Models', data: models }]}
      xScale={{ scale: 'log', domain: [1e6, 'auto'] }}
      yScale={{ scale: 'log', domain: [1e8, 'auto'] }}
      xAxisLabel="Parameters"
      yAxisLabel="FLOPs"
      aria-label="Model size vs FLOPs on log-log scale"
      height={300}
      useVoronoi
    />
  );
}
```

## Variant — bubble scatter with Z axis

```tsx preview
import { ScatterChart } from '@geti-ai/charts';

const points = [
  { x: 18, y: 0.72, z: 1200, name: 'Dataset A' },
  { x: 24, y: 0.81, z: 3400, name: 'Dataset B' },
  { x: 33, y: 0.87, z: 5200, name: 'Dataset C' },
  { x: 41, y: 0.90, z: 8600, name: 'Dataset D' },
];

export default function App() {
  return (
    <ScatterChart
      series={[{ name: 'Benchmarks', data: points, zKey: 'z' }]}
      zAxisProps={{ dataKey: 'z', range: [40, 400], name: 'Samples' }}
      xAxisLabel="Latency (ms)"
      yAxisLabel="F1 score"
      aria-label="Bubble scatter by dataset size"
      height={300}
      useVoronoi
    />
  );
}
```

## Variant — model families (category + bubble size)

```tsx preview
import { ScatterChart } from '@geti-ai/charts';

const yolo = [
  { x: 3.2e6, y: 0.41, z: 8.1, model: 'YOLO-N' },
  { x: 1.1e7, y: 0.49, z: 28.7, model: 'YOLO-S' },
  { x: 2.6e7, y: 0.54, z: 78.9, model: 'YOLO-M' },
];

const unet = [
  { x: 7.8e6, y: 0.44, z: 21.5, model: 'UNet-Tiny' },
  { x: 2.0e7, y: 0.51, z: 64.1, model: 'UNet-Base' },
  { x: 3.9e7, y: 0.56, z: 118.3, model: 'UNet-Large' },
];

const detr = [
  { x: 4.1e7, y: 0.52, z: 97.2, model: 'DETR-R50' },
  { x: 6.8e7, y: 0.57, z: 162.4, model: 'DETR-R101' },
];

export default function App() {
  return (
    <ScatterChart
      series={[
        { name: 'YOLO', data: yolo, xKey: 'x', yKey: 'y', zKey: 'z', shape: 'circle' },
        { name: 'UNet', data: unet, xKey: 'x', yKey: 'y', zKey: 'z', shape: 'diamond' },
        { name: 'DETR', data: detr, xKey: 'x', yKey: 'y', zKey: 'z', shape: 'triangle' },
      ]}
      zAxisProps={{ dataKey: 'z', range: [50, 460], name: 'GFLOPs' }}
      showLegend
      useVoronoi
      xScale={{ scale: 'log', domain: [1e6, 'auto'] }}
      xAxisLabel="Number of parameters"
      yAxisLabel="M5 COCO mAP @ 50:95"
      tooltipProps={{
        formatter: (value, name) => {
          if (name === 'z') return [`${Number(value).toFixed(1)} GFLOPs`, 'Complexity'];
          if (name === 'x') return [Number(value).toLocaleString(), 'Parameters'];
          return [value, name];
        },
        labelFormatter: (label) => String(label),
      }}
      aria-label="Model families by parameters, mAP, and complexity"
      height={340}
    />
  );
}
```

This pattern maps **category/family** to series (and therefore color/legend), while the `z` dimension controls bubble size.
