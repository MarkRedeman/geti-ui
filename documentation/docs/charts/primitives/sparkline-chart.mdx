# Sparkline chart (primitive)

`Sparkline` is exported from primitives and can be used as a compact inline primitive.

```tsx preview
import { Sparkline } from '@geti-ai/charts/primitives';

export default function App() {
  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: 12 }}>
      <Sparkline data={[72, 75, 79, 81, 83, 85, 88].map((value) => ({ value }))} area />
      <Sparkline data={[80, 82, 84, 86, 87, 89, 91].map((value) => ({ value }))} />
    </div>
  );
}
```
