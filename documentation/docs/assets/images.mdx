export async function copyImageImport(exportName) {
  const statement = `import { ${exportName} } from '@geti-ui/ui/assets/images';`;
  try {
    await navigator.clipboard.writeText(statement);
    toast.positive(`Copied import for ${exportName}`);
  } catch {
    toast.negative('Failed to copy import');
  }
}

export function ClickableCell({ name, children }) {
  return (
    <button
      type="button"
      onClick={() => copyImageImport(name)}
      aria-label={`Copy import for ${name}`}
      style={{
        ...CELL,
        width: '100%',
        cursor: 'copy',
        textAlign: 'inherit',
      }}
    >
      {children}
      <code style={LABEL}>{name}</code>
    </button>
  );
}

# Images

<ToastContainer placement="bottom" />

Illustration SVGs and background/loading webp images for common Geti UI states.

```tsx
import { WorkInProgressIcon, NotFound, ConnectionLost } from '@geti-ui/ui/assets/images';
import { LoginBackground, IntelLoading } from '@geti-ui/ui/assets/images';
```

## SVG illustrations

These are inline SVG React components that accept `SVGProps<SVGSVGElement>`. Render them directly with `width` / `height` props:

```tsx
import { WorkInProgressIcon, NotFound } from '@geti-ui/ui/assets/images';

<WorkInProgressIcon width={200} height={200} />
<NotFound width={180} height={180} />
```

## Static image URLs (webp)

These exports are URL strings — use them in `<img src={…}>` or CSS `background-image`. They are **not** React components.

```tsx
import { LoginBackground, IntelLoading, IntelAdminAppBackground, NoProjectsPlaceholder } from '@geti-ui/ui/assets/images';

// In JSX
<img src={LoginBackground} alt="Login background" style={{ width: '100%' }} />

// In CSS-in-JS
<div style={{ backgroundImage: `url(${LoginBackground})`, backgroundSize: 'cover' }} />
```
