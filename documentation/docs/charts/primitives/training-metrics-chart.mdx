# Training metrics chart (primitive)

Use Recharts `LineChart` primitives when you want full control over train/validation/test metric layers.

## Variant — loss curves with subset palette

```tsx preview
import { Line, LineChart } from 'recharts';
import {
  GetiCartesianGrid,
  GetiLegend,
  GetiResponsiveContainer,
  GetiTooltip,
  GetiXAxis,
  GetiYAxis,
} from '@geti-ai/charts/primitives';
import { getDatasetSubsetColor } from '@geti-ai/charts';

const data = [
  { epoch: 1, trainLoss: 1.42, valLoss: 1.53, testLoss: 1.63 },
  { epoch: 2, trainLoss: 1.08, valLoss: 1.20, testLoss: 1.31 },
  { epoch: 3, trainLoss: 0.86, valLoss: 0.98, testLoss: 1.08 },
  { epoch: 4, trainLoss: 0.70, valLoss: 0.83, testLoss: 0.92 },
  { epoch: 5, trainLoss: 0.58, valLoss: 0.72, testLoss: 0.79 },
  { epoch: 6, trainLoss: 0.49, valLoss: 0.64, testLoss: 0.70 },
];

export default function App() {
  return (
    <GetiResponsiveContainer height={340}>
      <LineChart data={data} margin={{ top: 8, right: 16, bottom: 8, left: 0 }}>
        <GetiCartesianGrid />
        <GetiXAxis dataKey="epoch" />
        <GetiYAxis />
        <GetiTooltip />
        <GetiLegend />

        <Line dataKey="trainLoss" name="Loss (Train)" stroke={getDatasetSubsetColor('train')} dot={false} isAnimationActive={false} />
        <Line dataKey="valLoss" name="Loss (Validation)" stroke={getDatasetSubsetColor('validation')} dot={false} strokeDasharray="5 3" isAnimationActive={false} />
        <Line dataKey="testLoss" name="Loss (Test)" stroke={getDatasetSubsetColor('test')} dot={false} strokeDasharray="2 3" isAnimationActive={false} />
      </LineChart>
    </GetiResponsiveContainer>
  );
}
```

## Variant — mixed metrics and custom formatter

```tsx preview
import { Line, LineChart } from 'recharts';
import {
  GetiCartesianGrid,
  GetiLegend,
  GetiResponsiveContainer,
  GetiTooltip,
  GetiXAxis,
  GetiYAxis,
} from '@geti-ai/charts/primitives';
import { getDatasetSubsetColor } from '@geti-ai/charts';

const data = [
  { epoch: 1, trainMap: 0.21, valMap: 0.18 },
  { epoch: 2, trainMap: 0.34, valMap: 0.30 },
  { epoch: 3, trainMap: 0.48, valMap: 0.43 },
  { epoch: 4, trainMap: 0.59, valMap: 0.54 },
  { epoch: 5, trainMap: 0.67, valMap: 0.62 },
  { epoch: 6, trainMap: 0.74, valMap: 0.69 },
];

export default function App() {
  return (
    <GetiResponsiveContainer height={320}>
      <LineChart data={data}>
        <GetiCartesianGrid />
        <GetiXAxis dataKey="epoch" />
        <GetiYAxis domain={[0, 1]} />
        <GetiTooltip formatter={(value) => [`${(Number(value) * 100).toFixed(1)}%`, 'score']} />
        <GetiLegend />

        <Line dataKey="trainMap" name="mAP (Train)" stroke={getDatasetSubsetColor('train')} dot={false} isAnimationActive={false} />
        <Line dataKey="valMap" name="mAP (Validation)" stroke={getDatasetSubsetColor('validation')} dot={false} strokeDasharray="5 3" isAnimationActive={false} />
      </LineChart>
    </GetiResponsiveContainer>
  );
}
```
